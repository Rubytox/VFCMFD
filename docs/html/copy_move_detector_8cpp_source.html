405 
406  string logString;
407 
408  InterestPoint pt = _interestPoints[i];
409  logString += "Looking for matches for InterestPoint:\n";
410  logString += infoKeypoint(pt);
411 
412  pair<int, int>&& rangeAngle = _interestPoints.getRangeAngle(pt);
413  int minIdxAngle = rangeAngle.first;
414  int maxIdxAngle = rangeAngle.second;
415 
416  logString += "Angle window size: " + to_string(maxIdxAngle - minIdxAngle) + " points\n";
417 
418  map<double, InterestPoint>&& similarityAngles = _interestPoints.similarityAngle(pt, minIdxAngle, maxIdxAngle);
419 
420  map<double, InterestPoint>::iterator it1, it2;
421  vector<InterestPoint> matchesAngles;
422  for (it1 = similarityAngles.begin(), it2 = next(it1);

428  matchesAngles.push_back(it1->second);

641 void copyMoveDetector::computeScores(double& sensitivity, double& specificity) const {
642  double TP = 0, FP = 0, FN = 0, TN = 0;
643 
644  /*
645  * We iterate over the pixels of both given mask and computed mask.
646  */
647  for (int x = 0; x < _mask.rows; x++) {
648  for (int y = 0; y < _mask.cols; y++) {
649  uchar reference = _mask.at<uchar>(x, y);
650  uchar computed = _computedMask.at<uchar>(x, y);
651 
652  if (reference == 0xFF && computed == 0xFF)
653  TP++;
654  else if (reference == 0 && computed == 0xFF)
655  FP++;
656  else if (reference == 0xFF && computed == 0)
657  FN++;
658  else
659  TN++;
660  }
661  }
662 
663  sensitivity = TP / (TP + FN);
664  specificity = TN / (TN + FP);
665 }

667 double shoelaceArea(const vector<Point2f>& points) {
668  double area = 0;
669 
670  int n = points.size() - 1;
671  area += points[n].x * points[0].y;
672  area -= points[0].x * points[n].y;
673 
674  for (int i = 0; i < n; i++) {
675  area += points[i].x * points[i + 1].y;
676  area -= points[i + 1].x * points[i].y;
677  }

712 Point2f pointFrom4D(int x1, int y1, double theta, double l) {
713  double sinTheta = sin(theta), cosTheta = cos(theta);
714  double sin2Theta = sinTheta * sinTheta, cos2Theta = cosTheta * cosTheta;
715  double rho = x1 * cosTheta + y1 * sinTheta;
716 
717  double A = l * l - x1 * x1 - y1 * y1 + 2 * y1 * rho / sinTheta - rho * rho / sin2Theta;
718  double B = cos2Theta / sin2Theta + 1;
719  double C = -2 * x1 + 2 * y1 * cosTheta / sinTheta - 2 * rho * cosTheta / sin2Theta;
720 
721  double Delta = C * C + 4 * A * B;
722 
723  double x21 = (-C - sqrt(Delta)) / (2 * B);
724  double x22 = (-C + sqrt(Delta)) / (2 * B);
725 
726  double x2 = x1 < x21 ? x21 : x22;
727  double y2 = (rho - x2 * cosTheta) / sinTheta;
728 
729  return Point2f(x2, y2);
730 }
